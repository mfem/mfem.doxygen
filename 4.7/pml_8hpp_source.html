75class PmlCoefficient : public Coefficient
76{
77private:
78 CartesianPML * pml = nullptr;
79 real_t (*Function)(const Vector &, CartesianPML * );
80public:
81 PmlCoefficient(real_t (*F)(const Vector &, CartesianPML *), CartesianPML * pml_)
82 : pml(pml_), Function(F)
83 {}
84 virtual real_t Eval(ElementTransformation &T, const IntegrationPoint &ip)
85 {
86 real_t x[3];
87 Vector transip(x, 3);
88 T.Transform(ip, transip);
89 return ((*Function)(transip, pml));
90 }
91};
