28// Define a HIP error check macro, MFEM_GPU_CHECK(x), where x returns/is of
29// type 'hipError_t'. This macro evaluates 'x' and raises an error if the
30// result is not hipSuccess.
31#define MFEM_GPU_CHECK(x) \
32 do \
33 { \
34 hipError_t err = (x); \
35 if (err != hipSuccess) \
36 { \
37 mfem_hip_error(err, #x, _MFEM_FUNC_NAME, __FILE__, __LINE__); \
38 } \
39 } \
40 while (0)
41#endif // MFEM_USE_HIP
