67template <class T>
68T Array<T>::Max() const
69{
70 MFEM_ASSERT(size > 0, "Array is empty with size " << size);
71
72 T max = operator[](0);
73 for (int i = 1; i < size; i++)

126template <class T>
127int Array<T>::IsSorted() const
128{
129 T val_prev = operator[](0), val;
130 for (int i = 1; i < size; i++)
131 {
132 val=operator[](i);
133 if (val < val_prev)
134 {
135 return 0;
136 }
137 val_prev = val;
138 }
139
140 return 1;
141}
