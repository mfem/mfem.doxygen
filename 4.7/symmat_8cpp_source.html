32void DenseSymmetricMatrix::SetSize(int s)
33{
34 MFEM_ASSERT(s >= 0,
35 "invalid DenseSymmetricMatrix size: " << s);
36 if (Height() == s)
37 {
38 return;
39 }
40 height = s;
41 width = s;
42 const int s2 = (s*(s+1))/2;
43 if (s2 > data.Capacity())
44 {
45 data.Delete();
46 data.New(s2);
