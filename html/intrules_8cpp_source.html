00051 void IntegrationRule::GaussianRule()
00052 {
00053    int n = NPoints;
00054    int m = (n+1)/2;
00055    int i, j;
00056    double p1, p2, p3;
00057    double pp, z, z1;
00058    for (i = 1; i <= m; i++)
00059    {
00060       z = cos ( M_PI * (i - 0.25) / (n + 0.5));
00061 
00062       while(1)
00063       {
00064          p1 = 1;
00065          p2 = 0;
00066          for (j = 1; j <= n; j++)
00067          {
00068             p3 = p2;
00069             p2 = p1;
00070             p1 = ((2 * j - 1) * z * p2 - (j - 1) * p3) / j;
00071          }
00072          // p1 is Legendre polynomial
00073 
00074          pp = n * (z*p1-p2) / (z*z - 1);
00075          z1 = z;
00076          z = z1-p1/pp;
00077 
00078          if (fabs (z - z1) < 1e-14) break;
00079       }
00080 
00081       IntPoints[i-1].x  = 0.5 * (1 - z);
00082       IntPoints[n-i].x  = 0.5 * (1 + z);
00083       IntPoints[i-1].weight = IntPoints[n-i].weight =
00084          1.0 / ( (1  - z * z) * pp * pp);
00085    }
00086 }
