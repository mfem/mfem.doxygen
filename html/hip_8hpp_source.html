28 // type 'hipError_t'. This macro evaluates 'x' and raises an error if the
29 // result is not hipSuccess.

31  do \
32  { \
33  hipError_t err = (x); \
34  if (err != hipSuccess) \
35  { \
36  mfem_hip_error(err, #x, _MFEM_FUNC_NAME, __FILE__, __LINE__); \
37  } \
38  } \
39  while (0)
